class listingpageform(forms.Form):
    
    

    bid=forms.DecimalField(label="Base Bid")



    def __init__(self, *args, min_value=0, **kwargs):
        super().__init__(*args, **kwargs)
        bid = self.fields['bid']
        bid.min_value = min_value+1
        if min_value is not None:
            bid.validators.append(MinValueValidator(min_value+1))
            bid.widget.attrs['min'] = min_value+1
